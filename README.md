import math
def calculate_distance(x1, y1, x2, y2):
    distance = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
    return round(distance, 2)
x1,y1 = 4, 2
x2,y2= 10, 6
bridge_length = calculate_distance(x1, y1,x2, y2)
print(f"{bridge_length}")
